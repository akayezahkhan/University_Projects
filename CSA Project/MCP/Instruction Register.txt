module Instruction_reg(
	input clk,
	input rst,
	input IRWrite,
	input [31:0] in,
	output[31:0] IR_out
    );
	 
	 reg [31:0] IR;
	 
	 always@(posedge clk or posedge rst)
		begin
			if (rst == 1)
				IR <= 32'b 000000_00010_00001_00000_00000_100000 ;	// first instruction
			else if (IRWrite == 1)
				IR <= in;
		end
		
	assign IR_out = IR;	
	
endmodule
